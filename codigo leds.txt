int pinDATA = 2; //DEFINIR EL PIN DE ENTRADA DEL SERIAL POR DONDE ENTRARA EL VALOR 1 O 0... DEFINE SI ES 1 O 0
int pinCLOCK = 5; //DEFINIR EL PIN USADO PARA EL RELOJ(the shift register clock input) GUARDA EL 1 0 0 QUE SE TOMA DE LA DATA
int pinLATCH = 4;//DEFINIR EL PIN QUE VA A CARGAR(storage register clock input) INVIA LOS DATOS GUARDADOS
String fila = "";
int tama単o = 0; //tama単o del arreglo para guardar todos los datos (crece)
String banco_de_datos [tama単o][8];
void setup() { 
  Serial.begin(9600);
  pinMode(pinDATA, OUTPUT);
  pinMode(pinLATCH, OUTPUT);
  pinMode(pinCLOCK, OUTPUT);
  digitalWrite(pinCLOCK,LOW);   
  digitalWrite(pinLATCH,LOW);   
}

void loop()
{
pedirDatos();
}


void pedirDatos()
{
  if(Serial.available()> 0){	
    String fila = "";
    delay(1000);
    while(Serial.available()> 0){
      fila += (char)Serial.read();
    }
    int lugar = 0;  			//en cual posicion del vector actual guardar
    banco_de_datos [tama単o][lugar] = fila;
    lugar ++;
  }
}
